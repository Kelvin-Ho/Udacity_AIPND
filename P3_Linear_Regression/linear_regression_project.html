def matxMultiply(A, B):
    # AB
    A_col = len(A[0])
    B_row = len(B)
    
    # 0 mul
    mul = [[0]*len(B[0]) for i in range(len(A))]
    
    # Try Catch
    if A_col == B_row:
        for i in range(len(A)):
            for j in range(len(B[0])):
                for n in range(len(A[0])):
                    mul[i][j] = mul[i][j] + A[i][n] * B[n][j]
        return mul
    else:
        raise ValueError

def augmentMatrix(A, b):
    Ab = [[0]*(len(A[0])+1) for i in range(len(A))]
    for i in range(len(A)):
        for j in range(len(A[0])):
            Ab[i][j] = A[i][j]
        Ab[i][len(A[0])] = b[i][0]   
    return Ab

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    
    if len(A) == len(b):
        Ab = augmentMatrix(A, b)
        matxRound(Ab, decPts)
        # 
        for n in range(len(A)):
            max_val = 0 #
            current_row = n # 
            for r in range(n,len(A)): # 
                if abs(Ab[r][n]) > max_val:
                    max_val = abs(Ab[r][n])
                    larger_row = r
            if max_val < epsilon: # 0 A
                return None
            else:
                swapRows(Ab, current_row, larger_row)
                #     
                scaleRow(Ab, n, 1.0 / Ab[n][n])
                for r in range(n):
                    addScaledRow(Ab, r, n, -Ab[r][n])
                for r in range(n+1, len(A)):
                    addScaledRow(Ab, r, n, -Ab[r][n])
        # bx
        x_matrix = []
        for i in range(len(Ab)):
            x_matrix.append([Ab[i][-1]])
        return x_matrix
    else: # Ab
        raise ValueError

# TODO MSE
def calculateMSE2D(X,Y,m,b):
    num_points = len(X)
    res = 0
    for i in range(num_points):
        res = (Y[i] - m * X[i] - b) ** 2 + res
    mse = res / num_points
    return mse

def linearRegression2D(X,Y):
    #  Ax = b , A=XTX, b=XTY, hm,b
    X = [[x, 1] for x in X]
    Y = [[y] for y in Y]
    A = matxMultiply(transpose(X), X)
    b = matxMultiply(transpose(X), Y)
    h = gj_Solve(A, b)
    return h[0][0], h[1][0]
